406  def _findGPSindex(self, GPS):
407  # Figure out how 'wide' each range is
408  XleftSpan = np.amax(self.X) - np.amin(self.X)
409  XrightSpan = self.vman.grid_res[0]
410  YleftSpan = np.amax(self.Y) - np.amin(self.Y)
411  YrightSpan = self.vman.grid_res[1]
412 
413  # Convert the left range into a 0-1 range (float)
414  idx = int((GPS[0] - np.amin(self.X))* XrightSpan / float(XleftSpan))
415  idy = int((GPS[1] - np.amin(self.Y))* YrightSpan / float(YleftSpan))
416  # Is it a valid GPS point?
417  if math.isnan(self.score_map[idx][idy]) or idx<0 or idy<0:
418  # if not, let the user know
419  print("Error: GPS point is out of bounds")
420  return [None, None]
421  else:
422  # otherwise, send back the coordinates
423  return [idx, idy]

503  def _greedyStep(self, UAV):
504  # just for shorthand notation
505  x = UAV.IDXplan[len(UAV.IDXplan)-1][0]
506  y = UAV.IDXplan[len(UAV.IDXplan)-1][1]
507 
508  # first check to make sure the transition is valid
509  if x>=self.vman.grid_res[0] or \
510  y>=self.vman.grid_res[1] or \
511  x<0 or y<0:
512  return None
513  # set an unobtainably low maximum value
514  max = -10000000
515  # set the direction to zero just for starters
516  dir = 0
517  # if there isn't a transition list for this node
518  # then the node doesn't exist, and any move is invalid
519  try:
520  # just for shorthand
521  Xition = self.X_map[x][y].Xitions
522  except:
523  # if that doesn't work, something is wrong. end the run
524  return None
525  # iterate through the 8 possible transitions
526  for i in range(8):
527  # check the indices of the transition
528  [I, J] = self._shiftVals(i)
529  # if the transition index is invalid an exception will be thrown
530  try:
531  # pull the score from the score map for the transition node
532  score = self.score_map[x+I][y+J]
533  # pull the UAV's mask value for the transition node
534  mask = UAV.plan_mask[x+I][y+J]
535  # pull the wave map value for the transition node
536  wave = UAV.wave_map[x+I][y+J]
537  # calculate the cost of changing the UAV's heading
538  headCost = abs(i*np.deg2rad(45)- \
539  (UAV.plan_heading[len(UAV.plan_heading)-1])*
540  np.deg2rad(45))
541  # normalize to a value between -PI and PI
542  if headCost > PI:
543  headCost = abs(headCost - 2 * PI)
544  # calculate the 2nd derivative of the score wrt the transition
545  d2S = Xition[i].dSscore-UAV.dSplan[len(UAV.dSplan)-1]
546  # calculate the reward of the move
547  reward = (UAV.scoreWt*score + \
548  UAV.dSwt*Xition[i].dSscore + \
549  UAV.d2Swt*d2S - \
550  UAV.windWt*Xition[i].thetaCost/PI - \
551  UAV.headWt*headCost/PI + \
552  UAV.waveWt*wave + \
553  UAV.timeWt*len(UAV.IDXplan)) * \
554  mask
555 
556  # we are trying to maximize the reward
557  if reward > max and \
558  x+I<self.vman.grid_res[0] and \
559  y+J<self.vman.grid_res[1] and \
560  x+I>=0 and y+J>=0:
561  # if the transition is valid and has the highest reward
562  # then track the reward score and the direction of the transition
563  max = reward
564  dir = i
565  # if the transition is invalid, skip it
566  except:
567  pass
568  # return the decision and the resulting score
569  return max, dir

578  Xk = deepcopy(self.vman.flowfield.u_field[:,:,self.plane])
579  # current velocity and direction estimates
580  vd_k = [[deepcopy(self.params.v0)],
581  [deepcopy(self.params.d0)]]

583  sens_mat = np.column_stack([self._sens_mat[0].flatten(), self._sens_mat[1].flatten()])
584  # apply the pseudoinverse
585  sens_mat_pinv = np.linalg.pinv(sens_mat)
586  # field actual
587  Xbar = deepcopy(self.Xbar)
588  # if the node is not masked by the UAV's path, use the estimate
589  for i in range(self.vman.grid_res[0]):
590  for j in range(self.vman.grid_res[1]):
591  if UAV.path_mask[i][j]==1:
592  Xbar[i][j]=Xk[i][j]
593  # convert Xk and Xbar to column vectors
594  Xbar = np.vstack(Xbar.flatten())
595  Xk = np.vstack(Xk.flatten())
596  # calculate the next estimate: Vdk+1 = Vdk + sens_mat_pinv*(Xbar-Xk)
597  vd_kp1=vd_k+np.matmul(sens_mat_pinv,Xbar-Xk)
598  # update the estimates
599  self.params.v0 = vd_kp1[0][0]
600  self.params.d0 = vd_kp1[1][0]
601  # update the wind speed estimate
602  self.WF['farm']['properties']['wind_speed'] = self.params.v0
603  # calculate the FLORIS model with the new speed
604  vmanTemp = VisualizationManager(self.WF, self.vman.grid_res)
605  # update the direction estimate
606  vmanTemp.flowfield.wind_direction = self.params.d0
607  # recalculate the FLORIS wake model with the new direction estimate
608  vmanTemp.flowfield.calculate_wake()
