155  # calculate the cost of moving through the wind at the transition heading

205  idx = [0,0]

213  # (such as for the initial wave map calculation)

216  # we test the score using the UAV's mask
217  if self.score_map[i][j]*UAV.plan_mask[i][j]>max:

221  if self.score_map[i][j]>max:
222  max=self.score_map[i][j]

321  def _findGPSindex(self, GPS):
322  # Figure out how 'wide' each range is
323  XleftSpan = np.amax(self.X) - np.amin(self.X)
324  XrightSpan = self.vman.grid_res[0]
325  YleftSpan = np.amax(self.Y) - np.amin(self.Y)
326  YrightSpan = self.vman.grid_res[1]
327 
328  # Convert the left range into a 0-1 range (float)
329  idx = int((GPS[0] - np.amin(self.X))* XrightSpan / float(XleftSpan))
330  idy = int((GPS[1] - np.amin(self.Y))* YrightSpan / float(YleftSpan))
331  # Is it a valid GPS point?
332  if math.isnan(self.score_map[idx][idy]) or idx<0 or idy<0:
333  # if not, let the user know
