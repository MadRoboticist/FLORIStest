250  page = [None] * int((self.params.Drange[1] -

252  self.params.Drange[2] + 1)

255  # which corresponds to the error plotted in the same 'page' index
256  Page = [None] * int((self.params.Drange[1] -
257  self.params.Drange[0]) /
258  self.params.Drange[2] + 1)
259  # MAX holds the largest error value
260  MAX = 0
261  # MIN holds the minimum error value over the given range
262  MIN = 0
263  # max holds the maximum u_field value over the given range
264  max = 0
265  # min holds the minimum u_field value over the given range
266  min = 0
267  while i <= self.params.Drange[1]:
268 
269  # set the new wind direction
270  ref.wind_direction = self.flowfield.wind_direction+np.deg2rad(i)
271  # a list of the coordinates of the turbines in the field
272  turbines = [turbine for _, turbine in ref.turbine_map.items()]
273  for k, turbine in enumerate(turbines):
274  turbine.yaw_angle = np.deg2rad(i)
275  ref.calculate_wake()

277  U2 = ref.u_field
278  # calculate the error
279  ref2.flowfield.u_field = U2-U1
280 
281  page[idx]=[ref2.flowfield.x[:, :, plane],
282  ref2.flowfield.y[:, :, plane],
283  ref2.flowfield.u_field[:, :, plane]]

285  ref.y[:, :, plane],
286  ref.u_field[:, :, plane]]
287  if np.amax(page[idx]) > MAX:
288  MAX = np.amax(page[idx][2])
289  elif np.amin(page[idx]) < MIN:
290  MIN = np.amin(page[idx][2])
291  if np.amax(Page[idx]) > max:

293  elif np.amin(Page[idx][2]) < min:
294  min = np.amin(Page[idx][2])

764  # calculate f(vbar,dbar) = xBar
765  print('vBar = ' + str(self.params.vBar))
766  print('dBar = ' + str(self.params.dBar))
767  vdBar = [[self.params.vBar], [self.params.dBar]]
768  self.WF['farm']['properties']['wind_speed'] = self.params.vBar
769  ff_bar_vman = VisualizationManager(self.WF, self.grid_res)
770  ff_bar_vman.flowfield.wind_direction = self.params.dBar
771  ff_bar_vman.flowfield.calculate_wake()
772  plane = int(ff_bar_vman.flowfield.grid_resolution.z * self.params.percent_height)
773  X = deepcopy(ff_bar_vman.flowfield.x[:,:,plane])
774  Y = deepcopy(ff_bar_vman.flowfield.y[:,:,plane])
775  ff_bar = deepcopy(ff_bar_vman.flowfield.u_field)
776  ff = deepcopy(ff_bar[:,:,plane])
777  xBar = np.vstack(ff.flatten()) # 1xn
778 
779  # initialize error
780  vdErr = [[self.params.vBar - self.params.v0],
781  [self.params.dBar - self.params.d0]] # initial error
782  speedError = list() # list for history of speed error
783  directionError = list() # list for history of direction error
784  print('Initial speed error = ' + str(vdErr[0]))
785  print('Initial direction error = ' + str(vdErr[1]))
786  speedError.append(vdErr[0][0])
787  directionError.append(vdErr[1][0])
788 
789  # set up first iteration
790  iters = 0
791  iterations = list() # iteration list for graphing
792  iterations.append(iters)
793  V_k = list() # list for history of vk
794  D_k = list() # list for history of dk
795  V_k.append(self.params.v0)
796  D_k.append(self.params.d0)
797  vd_k = [[self.params.v0], [self.params.d0]] # initial estimate
798  err_field = list()
799  masks = list()
800  errMax = 0
801 
802  while iters < self.params.iterMax and \
803  ( abs(speedError[iters]) > self.params.spErrMax or \
804  abs(directionError[iters]) > self.params.dirErrMax ):
805  # set new wind speed estimate for floris model
806  self.WF['farm']['properties']['wind_speed'] = vd_k[0][0]
807  # calculate f(vk,dk) = temp_x_k
808  temp0_vman = VisualizationManager(self.WF, self.grid_res)
809  temp0_vman.flowfield.wind_direction = vd_k[1][0] # new wind direction estimate
810  temp0_vman.flowfield.calculate_wake() # recalculate field for new wind direction
811  temp_x_k = deepcopy(temp0_vman.flowfield.u_field)
812  err_field.append(abs(ff_bar[:,:,plane]-temp_x_k[:,:,plane]))
813  if np.amax(err_field[iters]) > errMax:
814  errMax = np.amax(err_field[iters])
815  # print vk,dk as new speed and direction estimates
816  print('vk = ' + str(vd_k[0][0]))
817  print('dk = ' + str(vd_k[1][0]))
818 
819  # calculate f(vk+ev,dk) = temp_x_ev
820  self.WF['farm']['properties']['wind_speed'] = vd_k[0][0] + self.params.epSpeed # add speed epsilon to current estimate
821  temp1_vman = VisualizationManager(self.WF, self.grid_res)
822  temp1_vman.flowfield.wind_direction = vd_k[1][0] # set wind direction to current estimate
823  temp1_vman.flowfield.calculate_wake()
824  temp_x_ev = deepcopy(temp1_vman.flowfield.u_field)
825 
826  # calculate f(vk,dk+ed) = temp_x_ed
827  temp2_vman = deepcopy(temp0_vman) # we can use same flowfield from temp0
828  temp2_vman.flowfield.wind_direction = vd_k[1][0] + self.params.epDir # set wind direction to current estimate + epsilon
829  temp2_vman.flowfield.calculate_wake() # recalculate wake for new wind speed
830  temp_x_ed = deepcopy(temp2_vman.flowfield.u_field)
831 
832  # calculate gradient
833  temp_df_dv = (temp_x_ev[:,:,plane] - temp_x_k[:,:,plane]) / self.params.epSpeed # partial of f WRT speed
834  temp_df_dd = (temp_x_ed[:,:,plane] - temp_x_k[:,:,plane]) / self.params.epDir # partial of f WRT direction
835  temp_grad_fk = np.column_stack([temp_df_dv.flatten(), temp_df_dd.flatten()]) # 2xn gradient of f (jacobian)
836  temp_sens_mat_pinv = np.linalg.pinv(temp_grad_fk)
837 
838  # calculate mask
839  temp_dMax = np.amax(np.abs(temp_sens_mat_pinv[1]))
840  temp_vMax = np.amax(np.abs(temp_sens_mat_pinv[0]))
841  temp_Zd = np.abs(temp_sens_mat_pinv[1] / temp_dMax)
842  temp_Zv = np.abs(temp_sens_mat_pinv[0] / temp_vMax)
843  temp_Z_mask = temp_Zd * temp_Zv / np.amax(temp_Zd * temp_Zv)
844  # print(np.amax(temp_Z_mask))
845  temp_Z_mask = np.where(temp_Z_mask > self.params.mask_thresh, 1, 0)
846  # print(temp_Z_mask)
847  temp_sens_mat_pinv[0] = temp_sens_mat_pinv[0] * temp_Z_mask
848  temp_sens_mat_pinv[1] = temp_sens_mat_pinv[1] * temp_Z_mask
849  masks.append(temp_Z_mask)
850 
851  # calculate pseudoinverse[gradient{f(vk,dk)}]*{xBar-f(vk,dk)} = adj_vd (adjustment to current v&d estimates)

854  # calculate v_k+1 and d_k+1
855  vd_kp1 = vd_k + adj_vd
