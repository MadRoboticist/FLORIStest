242  self.X_map[i][j].score = self.score_map[i][j]

246  self.X_map[i][j].Xitions[k].thetaCost = k*np.deg2rad(45)-self.d0

252  # if the index is out of bounds, this will throw an exception
253  try:
254  # compute the first derivative of score with respect to transition

411  def _findGPSindex(self, GPS):
412  # Figure out how 'wide' each range is
413  XleftSpan = np.amax(self.X) - np.amin(self.X)
414  XrightSpan = self.vman.grid_res[0]
415  YleftSpan = np.amax(self.Y) - np.amin(self.Y)
416  YrightSpan = self.vman.grid_res[1]
417 
418  # Convert the left range into a 0-1 range (float)
419  idx = int((GPS[0] - np.amin(self.X))* XrightSpan / float(XleftSpan))
420  idy = int((GPS[1] - np.amin(self.Y))* YrightSpan / float(YleftSpan))
421  # Is it a valid GPS point?
422  if math.isnan(self.score_map[idx][idy]) or idx<0 or idy<0:
423  # if not, let the user know
424  print("Error: GPS point is out of bounds")
425  return [None, None]

427  # otherwise, send back the coordinates

508  def _greedyStep(self, UAV):
509  # just for shorthand notation
510  x = UAV.IDXplan[len(UAV.IDXplan)-1][0]
511  y = UAV.IDXplan[len(UAV.IDXplan)-1][1]
512 
513  # first check to make sure the transition is valid
514  if x>=self.vman.grid_res[0] or \
515  y>=self.vman.grid_res[1]:
516  return None
517  # set an unobtainably low maximum value
518  max = -10000000
519  # set the direction to zero just for starters
520  dir = 0
521  # if there isn't a transition list for this node
522  # then the node doesn't exist, and any move is invalid
523  try:
524  Xition = self.X_map[x][y].Xitions
525  except:
526  return None
527  # iterate through the 8 possible transitions
528  for i in range(8):
529  # check the indices of the transition
530  [I, J] = self._shiftVals(i)
531  # if the transition index is invalid an exception will be thrown
532  try:
533  # pull the score from the score map for the transition node
534  score = self.score_map[x+I][y+J]
535  # pull the UAV's mask value for the transition node
536  mask = UAV.score_mask[x+I][y+J]
537  # pull the wave map value for the transition node
538  wave = UAV.wave_map[x+I][y+J]
539  # calculate the reward of the move
540  reward = (self.params.scoreWt*score + \
541  self.params.dSwt*Xition[i].dSscore - \
542  self.params.windWt*Xition[i].thetaCost/PI + \
543  self.params.waveWt*wave + \
544  self.params.timeWt*len(UAV.IDXplan)) * \
545  mask
546  # we are trying to maximize the reward
547  if reward > max and \
548  x+I<self.vman.grid_res[0] and \
549  y+J<self.vman.grid_res[1] and \
550  x+I>=0 and y+J>=0:
551  # if the transition is valid and has the highest reward
552  # then track the reward score and the direction of the transition
553  max = reward
554  dir = i
555  # if the transition is invalid, skip it
556  except:
557  pass
558  return max, dir
