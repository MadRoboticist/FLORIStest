85  def _validateJSON(self, json_dict, type_map):
86  """
87  Verifies that the expected fields exist in the json input file and
88  validates the type of the input data by casting the fields to
89  appropriate values based on the predefined type maps in
90 
91  _turbineProperties
92 
93  _wakeProperties
94 
95  _farmProperties
96 
97  inputs:
98  json_dict: dict - Input dictionary with all elements of type str
99 
100  type_map: dict - Predefined type map for type checking inputs
101  structured as {"property": type}
102  outputs:
103  validated: dict - Validated and correctly typed input property
104  dictionary
105  """
106 
107  validated = {}
108 
109  # validate the object type
110  if "type" not in json_dict:
111  raise KeyError("'type' key is required")
112 
113  if json_dict["type"] not in self._validObjects:
114  raise ValueError("'type' must be one of {}".format(", ".join(self._validObjects)))
115 
116  validated["type"] = json_dict["type"]
117 
118  # validate the description
119  if "description" not in json_dict:
120  raise KeyError("'description' key is required")
121 
122  validated["description"] = json_dict["description"]
123 
124  # validate the properties dictionary
125  if "properties" not in json_dict:
126  raise KeyError("'properties' key is required")
127  # check every attribute in the predefined type dictionary for existence
128  # and proper type in the given inputs
129  propDict = {}
130  properties = json_dict["properties"]
131  for element in type_map:
132  if element not in properties:
133  raise KeyError("'{}' is required for object type '{}'".format(element, validated["type"]))
134 
135  value,error = self._cast_to_type(type_map[element], properties[element])
136  if error is not None:
137  raise error("'{}' must be of type '{}'".format(element, type_map[element]))
138 
139  propDict[element] = value
140 
141  validated["properties"] = propDict
142 
143  return validated

146  """
147  Casts the string input to the type in typecast

149  inputs:

152  value: str - the input string to cast to 'typecast'
153 
154  outputs:
155  position 0: type or None - the casted value
